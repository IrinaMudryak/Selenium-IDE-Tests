Create Registration Page with widget in Auction Mode
open /dms/tools/make_a_page_2
waitForPageToLoad 30000
click xpath=//div[@id='map']//div[contains(@data-page, 'registration')]//div[contains(text(), 'Registration')]
click xpath=//div[@id='map']//div[contains(text(), 'Add New Page')]
click css=span.mapx-button-ico.widget
type id=library-keyword-filter Registration
sendKeys id=library-keyword-filter ${KEY_ENTER}
dragAndDropToObject css=div.ico-widget.widget-registration xpath=//div[contains(@class, 'ax-container')]
click css=span.mapx-button-ico.page
type xpath=(//input[@type='text'])[3] registration_auto
doubleClick xpath=//div[contains(@id, 'registration')]
select xpath=//div[@data-option='auction_mode']//select label=Yes (Standard)
click //button[@type='button']
click css=div.mapx-button-ico.save
click css=div.mapx-button-ico.activate
click css=div.ax-btn
