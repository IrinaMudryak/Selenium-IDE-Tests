Create Order Page with widget of Order Form
open /dms/tools/make_a_page_2
waitForPageToLoad 30000
click xpath=//div[@id='map']//div[contains(@data-page, 'order')]//div[contains(text(), 'Order')]
click xpath=//div[@id='map']//div[contains(text(), 'Add New Page')]
click css=span.mapx-button-ico.widget
type id=library-keyword-filter Order
sendKeys id=library-keyword-filter ${KEY_ENTER}
dragAndDropToObject css=div.ico-widget.widget-form_order xpath=//div[contains(@class, 'ax-container')]
click css=span.mapx-button-ico.page
type xpath=(//input[@type='text'])[3] order_auto
click css=div.mapx-button-ico.save
click css=div.mapx-button-ico.activate
click css=div.ax-btn
