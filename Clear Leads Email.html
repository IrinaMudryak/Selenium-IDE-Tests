Clear Leads Email
store 1 loop
open /dms/settings/website#leads-email
waitForElementPresent //tr[${loop}]/td[2]/div/span/div 30000
while storedVars.loop<20
storeText //tr[${loop}]/td[2]/div/span/div mail
storeEval if (storedVars.mail !="") {nextMove="edit"} else  {nextMove="nextlist"; } nextMove
echo ${nextMove}
echo ${loop}
gotolabel ${nextMove}
label edit
click //tr[${loop}]//span[contains(@class, 'edit_btn')]
type //tr[${loop}]//input[contains(@name, '[email]')]
click //tr[${loop}]//span[contains(@class, 'save_btn')]//a[contains(@class, 'b_save')]
storeText //tr[${loop}]/td[2]/div/span/div mail
storeEval if (storedVars.mail !="") {nextMove="edit"} else  {nextMove="nextlist"; } nextMove
gotolabel ${nextMove}
label nextlist
store javascript{storedVars.loop ++}
endWhile
